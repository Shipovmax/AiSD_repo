## 2. Декораторы — полное руководство

### 2.1. Базовый шаблон

```python
def my_decorator(func):
    def wrapper(*args, **kwargs):
        # Код ДО
        print("До вызова")

        result = func(*args, **kwargs)  # Вызов функции

        # Код ПОСЛЕ
        print("После вызова")

        return result
    return wrapper

@my_decorator
def say_hello(name):
    print(f"Hello, {name}!")

# say_hello = my_decorator(say_hello)  # То же самое
```

### 2.2. Как писать декораторы — пошагово

**Шаг 1:** Функция внутри функции

```python
def outer(func):
    def inner():
        print("До")
        func()
        print("После")
    return inner  # Возвращаем функцию, не вызываем!
```

**Шаг 2:** Передача аргументов

```python
def decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper
```

**Шаг 3:** Сохранение метаданных

```python
import functools

def decorator(func):
    @functools.wraps(func)  # Сохраняет __name__, __doc__
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper
```

### 2.3. Практические примеры

#### Замер времени

```python
import time
import functools

def timing(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        end = time.perf_counter()
        print(f"{func.__name__} выполнилась за {end - start:.4f} сек")
        return result
    return wrapper

@timing
def slow_function():
    time.sleep(1)
    return "Готово"
```

#### Логирование

```python
def logged(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"Вызов {func.__name__}: args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} вернул: {result}")
        return result
    return wrapper
```

#### Кэширование (мемоизация)

```python
def cache(func):
    cache_store = {}

    @functools.wraps(func)
    def wrapper(*args):
        if args not in cache_store:
            cache_store[args] = func(*args)
        return cache_store[args]
    return wrapper

@cache
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

fibonacci(100)  # Мгновенно!
```

#### Повторные попытки (retry)

```python
def retry(max_attempts=3, delay=1):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt < max_attempts - 1:
                        time.sleep(delay)
            raise last_exception
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unstable_api():
    # Может упасть с вероятностью 70%
    pass
```

#### Проверка прав доступа

```python
def require_permission(permission):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(user, *args, **kwargs):
            if permission not in user.permissions:
                raise PermissionError(f"No {permission} permission")
            return func(user, *args, **kwargs)
        return wrapper
    return decorator

@require_permission('admin')
def delete_user(user, target_user):
    target_user.delete()
```

#### Rate Limiter

```python
from collections import defaultdict

def rate_limit(calls_per_second):
    min_interval = 1.0 / calls_per_second
    last_called = defaultdict(float)

    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            elapsed = time.time() - last_called[func.__name__]
            if elapsed < min_interval:
                time.sleep(min_interval - elapsed)
            last_called[func.__name__] = time.time()
            return func(*args, **kwargs)
        return wrapper
    return decorator

@rate_limit(calls_per_second=2)  # Не больше 2 в секунду
def api_call():
    print("API call")
```

## 5. Монады и функторы

### 5.1. Функтор

Объект, к которому можно применить `map`, не извлекая значение:


### 5.2. Maybe монада

```python
from typing import Generic, TypeVar, Optional, Callable

T = TypeVar('T')

class Maybe(Generic[T]):
    def __init__(self, value: Optional[T]):
        self._value = value

    @classmethod
    def just(cls, value: T) -> 'Maybe[T]':
        return cls(value)

    @classmethod
    def nothing(cls) -> 'Maybe[T]':
        return cls(None)

    def map(self, func: Callable[[T], T]) -> 'Maybe[T]':
        if self._value is None:
            return Maybe.nothing()
        return Maybe.just(func(self._value))

    def flat_map(self, func: Callable[[T], 'Maybe[T]']) -> 'Maybe[T]':
        if self._value is None:
            return Maybe.nothing()
        return func(self._value)

    def get_or_else(self, default: T) -> T:
        return self._value if self._value is not None else default

# Пример
def safe_divide(a, b) -> Maybe[float]:
    if b == 0:
        return Maybe.nothing()
    return Maybe.just(a / b)

def safe_sqrt(x) -> Maybe[float]:
    if x < 0:
        return Maybe.nothing()
    return Maybe.just(x ** 0.5)

# Цепочка без проверок None!
result = (Maybe.just(100)
          .flat_map(lambda x: safe_divide(x, 4))
          .flat_map(lambda x: safe_sqrt(x))
          .get_or_else(-1))

print(result)  # 5.0
```

### 5.3. Either/Result монада

```python
from dataclasses import dataclass
from typing import Union, Generic, TypeVar, Callable

E = TypeVar('E')
T = TypeVar('T')

@dataclass
class Ok(Generic[T]):
    value: T

@dataclass
class Err(Generic[E]):
    error: E

Result = Union[Ok[T], Err[E]]

class ResultMonad(Generic[T, E]):
    def __init__(self, result: Result[T, E]):
        self._result = result

    @classmethod
    def ok(cls, value: T) -> 'ResultMonad[T, E]':
        return cls(Ok(value))

    @classmethod
    def err(cls, error: E) -> 'ResultMonad[T, E]':
        return cls(Err(error))

    def is_ok(self) -> bool:
        return isinstance(self._result, Ok)

    def map(self, func: Callable[[T], T]) -> 'ResultMonad[T, E]':
        if self.is_ok():
            return ResultMonad.ok(func(self._result.value))
        return ResultMonad.err(self._result.error)

    def flat_map(self, func: Callable[[T], 'ResultMonad[T, E]']) -> 'ResultMonad[T, E]':
        if self.is_ok():
            return func(self._result.value)
        return ResultMonad.err(self._result.error)

# Пример: валидация
def parse_int(s: str) -> ResultMonad[int, str]:
    try:
        return ResultMonad.ok(int(s))
    except ValueError:
        return ResultMonad.err(f"Не удалось распарсить '{s}'")

def positive(n: int) -> ResultMonad[int, str]:
    if n > 0:
        return ResultMonad.ok(n)
    return ResultMonad.err(f"Число {n} не положительное")

result = (parse_int("42")
          .flat_map(positive)
          .map(lambda x: x * 2))

print(result)  # Ok(84)
```

## 6. Каррирование

### 6.1. Что это?

Преобразование функции от многих аргументов в цепочку функций от одного аргумента:

```python
# Обычная
def add(a, b, c):
    return a + b + c

add(1, 2, 3)  # 6

# Каррированная
def add_curried(a):
    def inner1(b):
        def inner2(c):
            return a + b + c
        return inner2
    return inner1

add_curried(1)(2)(3)  # 6

# Частичное применение
add_5 = add_curried(5)
add_5(2)(3)  # 10
```

### 6.2. Декоратор curry

```python
import functools

def curry(func):
    @functools.wraps(func)
    def curried(*args, **kwargs):
        if len(args) + len(kwargs) >= func.__code__.co_argcount:
            return func(*args, **kwargs)
        return functools.partial(func, *args, **kwargs)
    return curried

@curry
def add(a, b, c):
    return a + b + c

add(1, 2, 3)    # 6
add(1)(2, 3)    # 6
add(1)(2)(3)    # 6
add(5)(3, 2)    # 10
```

## 7. Pattern matching

### 7.1. match/case (Python 3.10+)

```python
def handle_command(command: str):
    match command.split():
        case ["quit"]:
            return "Exiting..."
        case ["load", filename]:
            return f"Loading {filename}"
        case ["process", *items]:
            return f"Processing {len(items)} items"
        case _:
            return "Unknown command"
```

### 7.2. Сопоставление структур

```python
from dataclasses import dataclass

@dataclass
class Point:
    x: int
    y: int

@dataclass
class Circle:
    center: Point
    radius: int

def describe(shape):
    match shape:
        case Point(x=0, y=0):
            return "Начало координат"
        case Point(x=x, y=y):
            return f"Точка в ({x}, {y})"
        case Circle(center=Point(x=0, y=0), radius=r):
            return f"Окружность в центре, радиус {r}"
        case _:
            return "Неизвестная фигура"
```

```python
        case n if n == 0:
            return "Ноль"
```

## 8. ФП в реальных проектах

### 8.1. Функциональное ядро, императивная оболочка

```python
from dataclasses import dataclass, frozen
from typing import List, Optional
from enum import Enum

class OrderStatus(Enum):
    PENDING = "pending"
    PAID = "paid"
    SHIPPED = "shipped"

@dataclass(frozen=True)  # Неизменяемый!
class Order:
    id: str
    items: List[str]
    total: float
    status: OrderStatus

# === ЧИСТОЕ ЯДРО (без побочных эффектов) ===

def create_order(items: List[str], prices: List[float]) -> Order:
    return Order(f"order-{hash(tuple(items))}", items, sum(prices), OrderStatus.PENDING)

def apply_payment(order: Order, payment_id: str) -> Optional[Order]:
    if order.status != OrderStatus.PENDING:
        return None
    return Order(order.id, order.items, order.total, OrderStatus.PAID)

# === ИМПЕРАТИВНАЯ ОБОЛОЧКА (побочные эффекты здесь) ===

class OrderService:
    def __init__(self, db, payment_gateway):
        self.db = db
        self.gateway = payment_gateway

    def create_and_pay(self, items, prices, card):
        order = create_order(items, prices)           # Чистое
        payment_id = self.gateway.charge(order.total, card)  # Эффект
        paid_order = apply_payment(order, payment_id) # Чистое
        self.db.save(paid_order)                      # Эффект
        return paid_order
```

### 8.2. Event Sourcing

```python
from dataclasses import dataclass
from typing import List, Union

@dataclass(frozen=True)
class OrderCreated:
    order_id: str
    items: List[str]

@dataclass(frozen=True)
class PaymentReceived:
    order_id: str
    amount: float

@dataclass
class OrderState:
    status: str = "created"
    items: List[str] = None
    paid_amount: float = 0

# Чистая функция-редьюсер
def apply_event(state: OrderState, event) -> OrderState:
    if isinstance(event, OrderCreated):
        return OrderState(status="created", items=event.items)
    elif isinstance(event, PaymentReceived):
        return OrderState(status="paid", items=state.items, paid_amount=event.amount)
    return state

# Восстановление состояния
def rebuild_state(events: List) -> OrderState:
    state = OrderState()
    for event in events:
        state = apply_event(state, event)
    return state
```

## Ключевые принципы ФП

1. **Чистые функции** — одинаковые аргументы = одинаковый результат
2. **Нет побочных эффектов** — не менять внешнее состояние
3. **Нет изменяемости** — создавать новые значения, не менять старые
4. **Функции высшего порядка** — функции принимают/возвращают функции
5. **Композиция** — строить сложное из простого

> **Главная идея:** Программа = вызов функций. Всё.
